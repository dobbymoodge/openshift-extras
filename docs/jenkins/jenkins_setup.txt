* rhc app create -l admin -p admin -t jenkins-1.4 -a $YOUR_JENKINS
* modify
/var/lib/stickshift/$YOUR_JENKINS-$YOUR_DOMAIN/app-root/data/config.xml and
change the slave configuration.  Be sure to replace $YOUR_JENKINS-$YOUR_DOMAIN

  <slaves>
    <slave>
      <name>master</name>
      <description></description>
      <remoteFS>/var/lib/stickshift/$YOUR_JENKINS-$YOUR_DOMAIN/app-root/data/</remoteFS>
      <numExecutors>1</numExecutors>
      <mode>NORMAL</mode>
      <retentionStrategy class="hudson.slaves.RetentionStrategy$Always"/>
      <launcher class="hudson.slaves.CommandLauncher">
        <agentCommand>java -jar /var/lib/stickshift/$YOUR_JENKINS-$YOUR_DOMAIN/app-root/data/war/WEB-INF/slave.jar</agentCommand>
      </launcher>
      <label></label>
      <nodeProperties/>
    </slave>
   </slaves>
